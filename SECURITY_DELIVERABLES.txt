                        USAGE EXAMPLES
================================================================================

SSRF Protection:
  from proxywhirl.security import validate_proxy_url_safety
  
  safe, reason = validate_proxy_url_safety('http://127.0.0.1:8080')
  if not safe:
      print(f"Blocked: {reason}")  # Output: "Blocked: Loopback address blocked"

PBKDF2 Key Derivation:
  from proxywhirl.security import derive_key_pbkdf2, verify_pbkdf2_key
  
  password = "my_secure_password"
  key, salt = derive_key_pbkdf2(password)
  
  # Verify password later
  if verify_pbkdf2_key(password, key, salt):
      print("Password verified")

Credential Redaction:
  from proxywhirl.logging_config import configure_logging
  
  configure_logging(redact_credentials=True)  # Enables credential redaction
  # Now all logs will have credentials redacted

Input Validation:
  from proxywhirl.security import (
      validate_input_length,
      validate_input_port,
      validate_proxy_credentials
  )
  
  validate_input_length(user_input, max_length=500)  # Raises if too long
  validate_input_port(8080)  # Raises if out of range (1-65535)
